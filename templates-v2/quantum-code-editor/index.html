/**
* Quantum Code Editor — Core Editor Module
* Handles text editing, syntax highlighting, and collaboration.
*/
import { SyntaxHighlighter } from './syntax';
import { ThemeManager } from './themes';
import { KeyBindingEngine } from './keybindings';
import { Terminal } from './terminal';
import { GitIntegration } from './git';
import { AutoComplete } from './autocomplete';
import { SearchEngine } from './search';
import { SnippetManager } from './snippets';
interface EditorConfig {
theme: string;
fontSize: number;
tabSize: number;
wordWrap: boolean;
minimap: boolean;
lineNumbers: boolean;
bracketMatching: boolean;
autoSave: boolean;
collaboration: CollabConfig;
}
interface CollabConfig {
enabled: boolean;
serverUrl: string;
roomId: string;
userName: string;
userColor: string;
}
export class Editor implements Disposable {
private highlighter: SyntaxHighlighter;
private themeManager: ThemeManager;
private keybindings: KeyBindingEngine;
private terminal: Terminal;
private git: GitIntegration;
private autocomplete: AutoComplete;
private search: SearchEngine;
private snippets: SnippetManager;
private buffer: TextBuffer;
private cursor: CursorPosition;
private selection: Selection | null;
private history: UndoStack;
private collab: CollabSession | null;
constructor(config: EditorConfig) {
this.highlighter = new SyntaxHighlighter();
this.themeManager = new ThemeManager(config.theme);
this.keybindings = new KeyBindingEngine();
this.terminal = new Terminal({ shell: '/bin/zsh' });
this.git = new GitIntegration();
this.autocomplete = new AutoComplete();
this.search = new SearchEngine();
this.snippets = new SnippetManager();
this.buffer = new TextBuffer();
this.cursor = { line: 0, col: 0 };
this.selection = null;
this.history = new UndoStack(500);
this.collab = null;
this.init();
}
private async init(): Promise<void> {
await this.themeManager.load();
this.keybindings.register(this);
this.snippets.loadBuiltins();
this.render();
this.emit('ready');
}
public openFile(path: string): void {
const content = readFileSync(path, 'utf-8');
const lang = this.detectLanguage(path);
this.buffer = new TextBuffer(content);
this.highlighter.setLanguage(lang);
this.history.clear();
this.render();
this.emit('fileOpened', { path, lang });
}
public save(): void {
const content = this.buffer.toString();
writeFileSync(this.currentPath, content);
this.git.stageFile(this.currentPath);
this.emit('saved', { path: this.currentPath });
}
public dispose(): void {
this.terminal.kill();
if (this.collab) this.collab.disconnect();
this.removeAllListeners();
}
}
