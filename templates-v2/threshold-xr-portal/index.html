Developer docs that connect portal content to immersive runtime.

Use the same content model for web cards, 3D annotations, and XR session payloads.

threshold-session.tsx 
import { ThresholdPortal } from "@threshold/xr";
export function ProductGateway() {
  return (
    <ThresholdPortal
      project="service-kit"
      entryMode="progressive"
      fallback="layered-cards"
      onSessionReady={(session) => {
        session.loadObject("service-kit.glb");
        session.enableCaptions("field-repair-en");
      }}
    />
  );
}
